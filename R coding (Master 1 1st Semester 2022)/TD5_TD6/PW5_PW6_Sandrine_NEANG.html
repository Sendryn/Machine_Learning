Question 6 : write a function that returns the RMSE of two vectors

RMSE <- function(actual, predicted) {
  result = sqrt(mean(actual - predicted)^2)
  return(result)
}

accuracy <- function(actual, predicted){
  m <- table(actual,predicted)
  True_positive = m[1]
  False_negative = m[3]
  False_positive = m[2]
  True_negative = m[4]
  result = (True_positive + True_negative) / (True_positive + False_negative + False_positive + True_negative)
  return(result)
}
